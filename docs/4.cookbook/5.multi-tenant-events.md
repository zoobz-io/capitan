# Multi-Tenant Events

> **Pattern Illustration**: This recipe demonstrates multi-tenant routing patterns.
> Define the signals, fields, and business logic for your specific SaaS domain.

Tenant-specific event routing and isolation for SaaS applications.

## Quick Start

Route events by tenant in 30 seconds:

```go
package main

import (
    "context"
    "log"
    "github.com/zoobzio/capitan"
)

// 1. Define your signals
var UserAction = capitan.NewSignal("user.action", "User performed action")

// 2. Define fields (including tenant ID)
var (
    TenantID = capitan.NewStringKey("tenant_id")
    UserID   = capitan.NewStringKey("user_id")
)

// 3. Route by tenant
func main() {
    c := capitan.New()
    defer c.Shutdown()

    c.Hook(UserAction, func(ctx context.Context, e *capitan.Event) {
        tenantID := TenantID.Extract(e)
        userID := UserID.Extract(e)

        // Route to tenant-specific handler
        switch tenantID {
        case "tenant-1":
            log.Printf("Tenant 1: User %s performed action", userID)
            // Apply tenant 1 business rules
        case "tenant-2":
            log.Printf("Tenant 2: User %s performed action", userID)
            // Apply tenant 2 business rules
        default:
            log.Printf("Unknown tenant: %s", tenantID)
        }
    })

    // Emit tenant-specific events
    c.Emit(context.Background(), UserAction,
        TenantID.Field("tenant-1"),
        UserID.Field("user-123"),
    )
}
```

## Problem

In multi-tenant SaaS applications, you need to:
- Route events to tenant-specific handlers
- Isolate tenant data
- Track per-tenant metrics
- Apply tenant-specific business rules

## Solution

```go
package tenants

import (
    "context"
    "log"
    "sync"
    "time"

    "github.com/zoobzio/capitan"
)

// Signals for multi-tenant pattern
var (
    DataCreated = capitan.NewSignal("data.created", "Data created")
    DataUpdated = capitan.NewSignal("data.updated", "Data updated")
    DataDeleted = capitan.NewSignal("data.deleted", "Data deleted")
)

// Field keys for multi-tenant pattern
var (
    TenantID   = capitan.NewStringKey("tenant_id")
    DataID     = capitan.NewStringKey("data_id")
    DataType   = capitan.NewStringKey("data_type")
    UserID     = capitan.NewStringKey("user_id")
    TenantPlan = capitan.NewStringKey("tenant_plan")
)

// Additional signals for production considerations
var (
    TenantOnboarded  = capitan.NewSignal("tenant.onboarded", "Tenant onboarded")
    TenantOffboarded = capitan.NewSignal("tenant.offboarded", "Tenant offboarded")
    ProcessInUSEast  = capitan.NewSignal("process.us-east", "Process in US East region")
    ProcessInEUWest  = capitan.NewSignal("process.eu-west", "Process in EU West region")
)

// Tenant represents a customer in the SaaS application
type Tenant struct {
    ID       string
    Name     string
    Plan     string
    Settings map[string]any
}

// TenantCache provides tenant-specific caching (stub implementation)
type TenantCache struct {
    tenantID string
    data     map[string]any
    mu       sync.RWMutex
}

func NewTenantCache(tenantID string) *TenantCache {
    return &TenantCache{
        tenantID: tenantID,
        data:     make(map[string]any),
    }
}

func (tc *TenantCache) Invalidate(key string) {
    tc.mu.Lock()
    defer tc.mu.Unlock()
    delete(tc.data, key)
}

func (tc *TenantCache) Clear() {
    tc.mu.Lock()
    defer tc.mu.Unlock()
    tc.data = make(map[string]any)
}

// TenantMetrics tracks per-tenant metrics (stub implementation)
type TenantMetrics struct {
    tenantID   string
    eventCount map[string]int64
    dataCount  int64
    mu         sync.RWMutex
}

func NewTenantMetrics(tenantID string) *TenantMetrics {
    return &TenantMetrics{
        tenantID:   tenantID,
        eventCount: make(map[string]int64),
    }
}

func (tm *TenantMetrics) IncrementEventCount(signal string) {
    tm.mu.Lock()
    defer tm.mu.Unlock()
    tm.eventCount[signal]++
}

func (tm *TenantMetrics) GetDataCount() int64 {
    tm.mu.RLock()
    defer tm.mu.RUnlock()
    return tm.dataCount
}

func (tm *TenantMetrics) IncrementDataCount() {
    tm.mu.Lock()
    defer tm.mu.Unlock()
    tm.dataCount++
}

func (tm *TenantMetrics) DecrementDataCount() {
    tm.mu.Lock()
    defer tm.mu.Unlock()
    tm.dataCount--
}

// TenantEventRouter routes events to tenant-specific handlers
type TenantEventRouter struct {
    c        *capitan.Capitan
    tenants  map[string]*TenantHandler
    mu       sync.RWMutex
}

// TenantHandler processes events for a specific tenant
type TenantHandler struct {
    tenant   *Tenant
    cache    *TenantCache
    metrics  *TenantMetrics
}

func NewTenantEventRouter(c *capitan.Capitan) *TenantEventRouter {
    return &TenantEventRouter{
        c:       c,
        tenants: make(map[string]*TenantHandler),
    }
}

func (ter *TenantEventRouter) RegisterTenant(tenant *Tenant) {
    ter.mu.Lock()
    defer ter.mu.Unlock()

    handler := &TenantHandler{
        tenant:  tenant,
        cache:   NewTenantCache(tenant.ID),
        metrics: NewTenantMetrics(tenant.ID),
    }

    ter.tenants[tenant.ID] = handler
}

func (ter *TenantEventRouter) UnregisterTenant(tenantID string) {
    ter.mu.Lock()
    defer ter.mu.Unlock()

    if handler, ok := ter.tenants[tenantID]; ok {
        handler.cache.Clear()
        delete(ter.tenants, tenantID)
    }
}

func (ter *TenantEventRouter) Start(c *capitan.Capitan) {
    // Route data events by tenant
    c.Hook(DataCreated, ter.routeEvent)
    c.Hook(DataUpdated, ter.routeEvent)
    c.Hook(DataDeleted, ter.routeEvent)
}

func (ter *TenantEventRouter) routeEvent(ctx context.Context, e *capitan.Event) {
    tenantID := TenantID.Extract(e)

    ter.mu.RLock()
    handler, ok := ter.tenants[tenantID]
    ter.mu.RUnlock()

    if !ok {
        log.Printf("No handler for tenant %s", tenantID)
        return
    }

    // Route to tenant-specific handler
    handler.ProcessEvent(ctx, e)
}

// TenantHandler methods

func (th *TenantHandler) ProcessEvent(ctx context.Context, e *capitan.Event) {
    // Track metrics
    th.metrics.IncrementEventCount(e.Signal().Name())

    // Apply tenant-specific business rules
    switch e.Signal() {
    case DataCreated:
        th.handleDataCreated(ctx, e)
    case DataUpdated:
        th.handleDataUpdated(ctx, e)
    case DataDeleted:
        th.handleDataDeleted(ctx, e)
    }
}

func (th *TenantHandler) handleDataCreated(ctx context.Context, e *capitan.Event) {
    dataID := DataID.Extract(e)

    // Check tenant limits
    if th.tenant.Plan == "free" && th.metrics.GetDataCount() >= 1000 {
        log.Printf("Tenant %s hit free plan limit", th.tenant.ID)
        // Emit limit exceeded event
        return
    }

    // Invalidate cache
    th.cache.Invalidate(dataID)

    // Track creation
    th.metrics.IncrementDataCount()

    log.Printf("[%s] Data created: %s", th.tenant.ID, dataID)
}

func (th *TenantHandler) handleDataUpdated(ctx context.Context, e *capitan.Event) {
    dataID := DataID.Extract(e)

    // Invalidate cache
    th.cache.Invalidate(dataID)

    log.Printf("[%s] Data updated: %s", th.tenant.ID, dataID)
}

func (th *TenantHandler) handleDataDeleted(ctx context.Context, e *capitan.Event) {
    dataID := DataID.Extract(e)

    // Remove from cache
    th.cache.Invalidate(dataID)

    // Track deletion
    th.metrics.DecrementDataCount()

    log.Printf("[%s] Data deleted: %s", th.tenant.ID, dataID)
}

// TenantCache

type TenantCache struct {
    tenantID string
    data     map[string]any
    mu       sync.RWMutex
}

func NewTenantCache(tenantID string) *TenantCache {
    return &TenantCache{
        tenantID: tenantID,
        data:     make(map[string]any),
    }
}

func (tc *TenantCache) Invalidate(key string) {
    tc.mu.Lock()
    defer tc.mu.Unlock()
    delete(tc.data, key)
}

func (tc *TenantCache) Clear() {
    tc.mu.Lock()
    defer tc.mu.Unlock()
    tc.data = make(map[string]any)
}

// TenantMetrics

type TenantMetrics struct {
    tenantID   string
    eventCount map[string]int64
    dataCount  int64
    mu         sync.RWMutex
}

func NewTenantMetrics(tenantID string) *TenantMetrics {
    return &TenantMetrics{
        tenantID:   tenantID,
        eventCount: make(map[string]int64),
    }
}

func (tm *TenantMetrics) IncrementEventCount(signal string) {
    tm.mu.Lock()
    defer tm.mu.Unlock()
    tm.eventCount[signal]++
}

func (tm *TenantMetrics) IncrementDataCount() {
    tm.mu.Lock()
    defer tm.mu.Unlock()
    tm.dataCount++
}

func (tm *TenantMetrics) DecrementDataCount() {
    tm.mu.Lock()
    defer tm.mu.Unlock()
    tm.dataCount--
}

func (tm *TenantMetrics) GetDataCount() int64 {
    tm.mu.RLock()
    defer tm.mu.RUnlock()
    return tm.dataCount
}
```

## Usage

```go
package main

import (
    "context"

    "github.com/zoobzio/capitan"
    "your-app/events"
    "your-app/fields"
    "your-app/tenants"
)

func main() {
    c := capitan.New()
    defer c.Shutdown()

    // Create tenant router
    router := tenants.NewTenantEventRouter(c)
    router.Start(c)

    // Register tenants
    router.RegisterTenant(&tenants.Tenant{
        ID:   "tenant-1",
        Name: "Acme Corp",
        Plan: "enterprise",
    })

    router.RegisterTenant(&tenants.Tenant{
        ID:   "tenant-2",
        Name: "Small Startup",
        Plan: "free",
    })

    // Emit tenant-specific events
    ctx := context.Background()

    c.Emit(ctx, DataCreated,
        TenantID.Field("tenant-1"),
        DataID.Field("data-123"),
    )

    c.Emit(ctx, DataCreated,
        TenantID.Field("tenant-2"),
        DataID.Field("data-456"),
    )

    c.Shutdown()
}
```

## Tenant Isolation

Ensure tenant data never crosses boundaries:

```go
// Tenant context key
type tenantKey struct{}

func WithTenant(ctx context.Context, tenantID string) context.Context {
    return context.WithValue(ctx, tenantKey{}, tenantID)
}

func GetTenant(ctx context.Context) string {
    if tenantID, ok := ctx.Value(tenantKey{}).(string); ok {
        return tenantID
    }
    return ""
}

// Middleware: Extract tenant from request
func TenantMiddleware(next http.Handler) http.Handler {
    return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
        tenantID := r.Header.Get("X-Tenant-ID")
        if tenantID == "" {
            http.Error(w, "Missing tenant ID", http.StatusBadRequest)
            return
        }

        ctx := WithTenant(r.Context(), tenantID)
        next.ServeHTTP(w, r.WithContext(ctx))
    })
}

// Use in event handlers
c.Hook(DataCreated, func(ctx context.Context, e *capitan.Event) {
    tenantID := GetTenant(ctx)

    // Verify tenant ID matches event
    eventTenantID := TenantID.Extract(e)
    if tenantID != eventTenantID {
        log.Printf("Tenant mismatch: ctx=%s, event=%s", tenantID, eventTenantID)
        return
    }

    // Process for correct tenant
    processData(tenantID, e)
})
```

## Per-Tenant Configuration

Apply tenant-specific settings:

```go
func (th *TenantHandler) ProcessEvent(ctx context.Context, e *capitan.Event) {
    // Apply tenant-specific webhooks
    if webhookURL, ok := th.tenant.Settings["webhook_url"].(string); ok {
        go sendWebhook(webhookURL, e)
    }

    // Apply tenant-specific rate limits
    if rateLimit, ok := th.tenant.Settings["rate_limit"].(int); ok {
        if th.metrics.GetEventCountInWindow(time.Minute) > int64(rateLimit) {
            log.Printf("Tenant %s exceeded rate limit", th.tenant.ID)
            return
        }
    }

    // Apply tenant-specific feature flags
    if featureEnabled, ok := th.tenant.Settings["feature_x_enabled"].(bool); ok && featureEnabled {
        enableFeatureX(ctx, e)
    }
}
```

## Production Considerations

### 1. Tenant Onboarding/Offboarding

```go
// Onboard new tenant
func (ter *TenantEventRouter) OnboardTenant(tenant *Tenant) {
    ter.RegisterTenant(tenant)

    // Emit onboarding event
    ter.c.Emit(context.Background(), TenantOnboarded,
        TenantID.Field(tenant.ID),
        TenantPlan.Field(tenant.Plan),
    )
}

// Offboard tenant
func (ter *TenantEventRouter) OffboardTenant(tenantID string) {
    ter.UnregisterTenant(tenantID)

    // Emit offboarding event
    ter.c.Emit(context.Background(), TenantOffboarded,
        TenantID.Field(tenantID),
    )
}
```

### 2. Tenant-Specific Metrics

```go
// Export metrics per tenant
go func() {
    ticker := time.NewTicker(10 * time.Second)
    defer ticker.Stop()

    for range ticker.C {
        ter.mu.RLock()
        for tenantID, handler := range ter.tenants {
            metrics.Gauge("tenant.data.count", float64(handler.metrics.GetDataCount()),
                "tenant_id", tenantID,
                "plan", handler.tenant.Plan,
            )
        }
        ter.mu.RUnlock()
    }
}()
```

### 3. Tenant Data Residency

```go
// Route to region-specific handlers
func (ter *TenantEventRouter) routeByRegion(ctx context.Context, e *capitan.Event) {
    tenantID := TenantID.Extract(e)

    ter.mu.RLock()
    handler, ok := ter.tenants[tenantID]
    ter.mu.RUnlock()

    if !ok {
        return
    }

    // Route to regional handler
    region := handler.tenant.Settings["region"].(string)
    switch region {
    case "us-east-1":
        ter.c.Emit(ctx, ProcessInUSEast, TenantID.Field(tenantID))
    case "eu-west-1":
        ter.c.Emit(ctx, ProcessInEUWest, TenantID.Field(tenantID))
    }
}
```
